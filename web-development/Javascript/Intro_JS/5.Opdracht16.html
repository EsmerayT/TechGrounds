            //charAt()
            let text = "Hello";
            let result = text.charAt(0);
            console.log(result); 
            //H

            Als er geen index is gegeven aan charAt(), dan is de standaard 0.
            Met 0 wordt het eerste character weergeven, dus H.
            charAt(1) index 1 weergeeft character e.


            //concat()
            let string = "Hello";
            let text = "world!";
            console.log(string.concat(text)); 
            //Hello world!

            De concat () methode wordt gebruikt om twee of meer strings samen te voegen.

            //slice()
            let string = "Hello world!";
            console.log(string.slice(0, 5));
            //Hello

            De methode slice () extraheert delen van een string en retourneert de geëxtraheerde delen in een nieuwe string.

            //split()
            let string = "How are you doing today?";
            console.log(string.split(" ")); 
            //(5) ["How", "are", "you", "doing", "today?"]

            De methode split () splitst een string in een lijst.

            //substr()
            let string = "Hello world!";
           console.log(string.substr(1, 4));
           //ello

           De methode substr () extraheert delen van een tekenreeks, beginnend bij het teken op de opgegeven positie,
           en retourneert het opgegeven aantal tekens.
