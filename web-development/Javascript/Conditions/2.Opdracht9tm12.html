function checkNumber() {
            let number = document.getElementById("number").value;
            if (number > 5 && number < 10) {
                document.getElementById("label").innerHTML = "Het ingevoerde getal is tussen 5 en 10"; //Opdracht 9
            } if (number >= 11 && number <= 20) {
                document.getElementById("label").innerHTML = "Het ingevoerde getal is gelijk aan 11 of groter dan 11 en is gelijk aan 20 of kleiner dan 20"; //Opdracht 10
            } if (number == 21 || number == 23) {
                document.getElementById("label").innerHTML = "Het ingevoerde getal is gelijk aan 21 of is gelijk aan 23"; //Opdracht 11
            } if (number > 30  && number < 35 || number > 40 && number < 45 )
                document.getElementById("label").innerHTML = "het getal is tussen 30 en 35 of het getal is tussen 40 en 45"; //Opdracht 12
        }
