      function checkNumber() {
        let number = (document.getElementById("number").value);
        if (number == 3) {
          document.getElementById("label").innerHTML = "Het ingevoerde getal is gelijk aan 3"; //Opdracht 1
        } else if (number < 3) {
          document.getElementById("label").innerHTML = "Het ingevoerde getal is kleiner dan 3"; //Opdracht 2
        } else if (number >= 11) {
          document.getElementById("label").innerHTML =
            "Het ingevoerde getal is gelijk of groter dan 11"; //Opdracht 3
        } else if (number > 4) {
          document.getElementById("label").innerHTML = "Het ingevoerde getal is groter dan 4"; //Opdracht 4
        } if (!number) {
          document.getElementById("label").innerHTML = ""; //Opdracht 7
        }
      }
